#include <SPI.h>
#include <MFRC522.h>

// RFID RC522 Pins
#define SS_PIN 10
#define RST_PIN 9

// Output Pins
#define BUZZER_PIN 5
#define GREEN_LED 6
#define RED_LED 7

MFRC522 rfid(SS_PIN, RST_PIN);

// Replace these UID values with your real RFID card UIDs
String START_UID  = "1702F006";   // Card 1 = START
String MOVE_UID   = "11223344";   // Card 2 = MOVE
String LEFT_UID   = "55667788";   // Card 3 = LEFT
String SUBMIT_UID = "AABBCCDD";   // Card 4 = SUBMIT

void setup() {
  Serial.begin(9600);

  SPI.begin();
  rfid.PCD_Init();

  pinMode(GREEN_LED, OUTPUT);
  pinMode(RED_LED, OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);

  digitalWrite(GREEN_LED, LOW);
  digitalWrite(RED_LED, LOW);
  digitalWrite(BUZZER_PIN, LOW);

  Serial.println("SYSTEM:TangiQuest Maze Game Ready");
  Serial.println("INFO:Scan START card to begin");
}

void loop() {
  if (!rfid.PICC_IsNewCardPresent()) {
    return;
  }

  if (!rfid.PICC_ReadCardSerial()) {
    return;
  }

  String uid = getUID();

  Serial.print("UID:");
  Serial.println(uid);

  if (uid == START_UID) {
    sendCommand("START");
  }
  else if (uid == MOVE_UID) {
    sendCommand("MOVE");
  }
  else if (uid == LEFT_UID) {
    sendCommand("LEFT");
  }
  else if (uid == SUBMIT_UID) {
    sendCommand("SUBMIT");
  }
  else {
    Serial.println("COMMAND:UNKNOWN");
    errorFeedback();
  }

  rfid.PICC_HaltA();
  rfid.PCD_StopCrypto1();

  delay(900);
}

String getUID() {
  String uid = "";

  for (byte i = 0; i < rfid.uid.size; i++) {
    if (rfid.uid.uidByte[i] < 0x10) {
      uid += "0";
    }

    uid += String(rfid.uid.uidByte[i], HEX);
  }

  uid.toUpperCase();
  return uid;
}

void sendCommand(String command) {
  Serial.print("COMMAND:");
  Serial.println(command);

  validFeedback();
}

void validFeedback() {
  digitalWrite(GREEN_LED, HIGH);
  digitalWrite(RED_LED, LOW);

  tone(BUZZER_PIN, 1000, 150);
  delay(200);

  digitalWrite(GREEN_LED, LOW);
}

void errorFeedback() {
  digitalWrite(RED_LED, HIGH);
  digitalWrite(GREEN_LED, LOW);

  tone(BUZZER_PIN, 300, 500);
  delay(600);

  digitalWrite(RED_LED, LOW);
}
